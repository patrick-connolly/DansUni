class ComplexNumber {

    private:
        double re{};
        double im{};
    
    public:
        ComplexNumber(double re = 0, double im = 0) : re(re), im(im) {}

        friend os::stream& operator<<(std::ostream& out, const ComplexNumber& num) {
            if(num.re == 0 && num.im == 0) {
                out << 0;
            }
            else if(num.re == 0 && num.im != 0) {
                out << num.im << "i";
            }
            else if(num.re != 0 && num.im == 0) {
                out << num.re;
            }
            else if(num.im > 0){
                out << num.re << " + " << num.im << "i"; 
            }
            else {
                out << num.re << " - " << -num.im << "i"; 
            }

            return out;
        }

        friend ComplexNumber operator+(const ComplexNumber& n1, const ComplexNumber& n2) {
            return { ComplexNumber( (n1.re + n2.re), (n1.im + n2.im) ) };
        } 

        friend ComplexNumber operator*(const ComplexNumber& n1, const ComplexNumber& n2) {
            double real = n1.re*n2.re - n1.im*n2.im;
            double im = n1.re*n2.im + n1.im*n2.re;
            return { ComplexNumber( (real), (im) ) };
        } 

        ComplexNumber pow(int n) {
            if(n == 0) {
                return { ComplexNumber(1,0)};
            }

            for(int i = 0; i < n; i++) {
                *this = *this * (*this);
            }

            return *this;
        }

        ComplexNumber& operator=(const ComplexNumber& num) {
            this->re = num.re;
            this->im = num.im;
            return *this;
        }

};


class RealNumber : public ComplexNumber {

    public:
        RealNumber(double re) : ComplexNumber(re, 0) {}
};

int main() {

    ComplexNumber* nums {new ComplexNumber[5]};

    for(int i = 0; i < 5, i++) {
        nums[i] = ComplexNumber(i+1, i+2);
    }

    ComplexNumber result;

    for(int i = 0; i < 5, i++) {
        result += nums[i];
    }

    std::cout << "Sum is: " << result << std::endl;

    ComplexNumber power(1,2);

    ComplexNumber lambda;

    for(int i = 0; i < 6, i++) {
        lambda += power.pow(i);
    }

    std::cout << "Lambda is: " << lambda << std::endl;

    delete[] nums;

    return 0;
}
