#include <iostream>
#include <string>
#include <memory>
#include <vector>

class Message {
private:
    std::string text;
    mutable int read_count { 0 };

public:
    Message(std::string text = "")
        : text { text }
    {
    }

    std::string get_text() const {
        ++read_count;
        return text;
    }

    void set_text(const std::string& new_text) {
        text = new_text;
    }

    int get_read_count() const {
        return read_count;
    }

    Message& operator=(const Message& other) {
        text = other.text;
        read_count = other.read_count;
        return *this;
    }

    friend std::ostream& operator<<(std::ostream& out, const Message& msg);
};

std::ostream& operator<<(std::ostream& out, const Message& msg) {
    out << msg.text;
    return out;
}

class Notification {
public:
    void identify() const {
        std::cout << "Generic notification\n";
    }

    virtual void send() const {
        std::cout << "Sending generic notification: ";
    }

    virtual ~Notification() = default;
};

class EmailNotification : public Notification {
private:
    Message message;

public:
    EmailNotification(const Message& message)
        : message { message }
    {
    }

    void identify() const {
        std::cout << "Email notification\n";
    }

    void send() const override {
        std::cout << "Sending email: " << message << '\n';
    }

    const Message& get_message() const {
        return message;
    }
};

template <typename T>
T choose_larger(const T& a, const T& b) {
    return (a > b) ? a : b;
}

std::string make_polite(const std::string& command) {
    return command + " please.";
}

std::string make_polite(std::unique_ptr<std::string> command) {
    return *command + " if you don't mind.";
}
