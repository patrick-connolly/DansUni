template<typename T>

class Pair {
    private:
        T first;
        T second;

    public:

        Pair(T first, T second) : first(first), second(second) {}

        T get_first() const {
            return first;
        }

        T get_second() const {
            return second;
        }

        void swap_values() {
            T temp = first;
            first = second;
            second = temp;
        }
};
