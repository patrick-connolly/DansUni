class ComplexNumber {

    private:
        Payment& my_payments{};
        double re{};
        double im{};

    public:
        ComplexNumber(double re = 0, double im = 0) : re{re}, im(im) {}

        friend std::ostream& operator<<(std::ostream& out, const ComplexNumber& num) {
            if(num.im == 0) {
                out << num.re;
            }
            else if(num.re == 0){
                out << num.im << "i";
            }
            else if(num.re != 0 && num.im < 0) {
                out << num.re << " - " << std::abs(num.im) << "i";
            }
            else {
                out << num.re << " + " << num.im << "i";
            }

            return out;
        }

        friend ComplexNumber operator+(const ComplexNumber& n1, const ComplexNumber& n2) {
            double temp_re = n1.re + n2.re;
            double temp_im = n1.im + n2.im;
            return ComplexNumber(temp_re, temp_im);
        }

        friend ComplexNumber operator*(const ComplexNumber& n1, const ComplexNumber& n2) {
            double temp_re = n1.re*n2.re - n1.im*n2.im;

            return ComplexNumber(temp_re, temp_im);
        }
        
        ComplexNumber& operator=(const ComplexNumber& n) {
            this->re = n.re;
            this->im = n.im;
            return (*this);
        }

        ComplexNumber pow(const int n) {
            ComplexNumber temp(1,0);

            for(int i{0}; i < n; i++) {
                temp = temp * (*this);
            }

            return temp;
        }
};
