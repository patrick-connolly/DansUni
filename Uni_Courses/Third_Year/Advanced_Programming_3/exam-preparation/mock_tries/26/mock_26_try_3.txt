class ComplexNumber {

    private:
        double re;
        double im;

    public:
        ComplexNumber(double re = 0, double im = 0) : re(re), im(im) {}

        friend std::ostream& operator<<(std::ostream& out, const ComplexNumber& n) {
            if(n.re == 0) {
                out << n.im << "i";
            }
            else if(n.im == 0) {
                out << n.re;
            }
            else if(n.im < 0) {
                out << n.re << " - " << std::abs(n.im) << "i";
            }
            else {
                out << n.re << " + " << std::abs(n.im) << "i";
            }

            return out;
        }

        friend ComplexNumber operator+(const ComplexNumber& n1, const ComplexNumber n2) {
            return ComplexNumber(n1.re+n2.re , n1.im + n2.im);
        }

        friend ComplexNumber operator*(const ComplexNumber& n1, const ComplexNumber n2) {
            double new_re = n1.re*n2.re - n1.im*n2.im;
            double new_im = n1.re*n2.im + n2.re*n1.im;
            return ComplexNumber(new_re, new_im);
        }

        ComplexNumber pow(int n) {
            ComplexNumber result(1,0);

            for(int i{0}; i < n; i++) {
                result = result * (*this);
            }

            return result;
        }

        ComplexNumber& operator=(const ComplexNumber& n) {
            this->re = n.re;
            this->im = n.im;
            return (*this);
        }

};

class RealNumber : public ComplexNumber {
    public:
        RealNumber(double re) : ComplexNumber(re, 0) {}
};

int main( {

    ComplexNumber* ptr {new ComplexNumber[5]};
    
    for(int i{0}; i < 5; i++) {
        ptr[i] = ComplexNumber(i+1, i+2);
    }

        sum = sum + ptr[i];
    }

    std::cout < "The sum is: " << sum << std::endl;
