    *ptr += 10;
}

// 2. Doubles the value of 'n' using a reference.
void double_by_reference(int& ref) {

// 3. Swaps the address held by two pointers (Pointer to Pointer).
// Note: This should change where the pointers in main point to.
void swap_pointers(int** ptr1, int** ptr2) {
    int* temp = *ptr1;
    *ptr1 = *ptr2;
    *ptr2 = temp;
}

class ComplexNumber {

    private:
        double re;
        double im;

    public:

        ComplexNumber(double re = 0, double im = 0) : re(re), im(im) {}
    
        friend std::ostream& operator<<(std::ostream& out, const ComplexNumber& n) {
            if(n.im == 0) {
                out << n.re;
            }
            else if(n.re == 0) {
                out << n.im;
            }
            else {
                out << n.re << " " << n.im << "i";
            }

            return out;
        }

        friend ComplexNumber operator+(const ComplexNumber& n1, const ComplexNumber& n2 ) {
            return ComplexNumber( n1.re+n2.re , n1.im + n2.im);
